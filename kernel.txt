%macro pinta 5 					; color, x, y, w, h
    pusha
    mov ax, %3
    mov cx, 320
    mul cx
    mov dx, %2
    add dx, ax
    mov di, dx
    mov cx, %5
    .line:
        push cx
        mov cx, %4
        .col:
            mov al, %1
            stosb 
        loop .col
        sub di, %4
        add di,320
        pop cx
    loop .line
    popa
%endmacro

;explicado na macro abaixo
;---------------------------------------------------------------------------------------------------------------------------------
%macro sprite 5  				; sprite, x, y, w, h   <----- POSIÇÕES (X,Y) ESPESSURA (W,H)
    pusha						;pra não perder nenhuma informação de registrador
    
    mov bx, %1      			;sprite
    mov ax, %3					;y
    mov cx, 320					
    mul cx						;multplica y por 320
    mov dx, %2					;x
    add dx, ax					;add x
    mov di, dx					;di = dx
    mov cx, %5					;cx = h
    .linha:						
    push cx						;h vai tá na pilha de cx
    mov cx, %4					;cx = wheight
            .coluna:
                    mov al,[bx]	;al vai receber a cor do pixel
                    stosb       ;ele stora os bytes em uma string, fazendo, [es:di] = al, que é a ideia 
                    inc bx		;vai para o próximo pixel
            loop .coluna		;vai executar o código de .coluna na quantidade que o reg cx tiver
            sub di, %4			;o que foi somado de x foi para di, agora subtraiu de w
            add di,320			;e adiciona 320 para ficar na posição x desejada
    pop cx						;volta a pilha pra h
    loop .linha					;a mesma coisa do "loop .coluna" só que com outro valor em cxs
    popa						; volta todos os registradores para o valor de antes da macro
%endmacro
